<CodeBlock>{`
function isPalindrome(S) {
  let iteration = Math.round(S.length / 2);
  let backward = S.length;
  for (let forward = 0; forward < iteration; forward++) {
    backward--;
    if (S[forward].toLowerCase() !== S[backward].toLowerCase()) {
      return "NO";
    }
  }
  return "YES";
}
`}</CodeBlock>
